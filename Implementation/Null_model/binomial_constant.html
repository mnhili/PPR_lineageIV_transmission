Model

Define Nimble code

ppr_code <- nimbleCode({
  # Prior : Uniforme
  p ~ dunif(0, 1)
  
  # Likelihood : Bernoulli
  for (i in 1:nrow(sero)) {
    y[i] ~ dbinom(size = n[i], prob = p)
  }
})

Define inits and data

y <- sero$Positives
n <- sero$Positives + sero$Negatives
data = list(y=y, n=n)

inits = lapply(1:10, function(i) {
  list(p = runif(1))
